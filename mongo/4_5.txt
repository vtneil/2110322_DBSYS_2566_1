db.customer.aggregate([
  {
    $lookup: {
      from: "orders_w_lines",
      localField: "customer_id",
      foreignField: "customer_id",
      as: "orders"
    }
  },
  {
    $project: {
      _id: 0,
      customer_id: 1,
      customer_name: 1,
      order_count: { $size: "$orders" }
    }
  },
  {
    $sort: {
      order_count: -1
    }
  }
])
