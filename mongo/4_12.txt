db.orders_w_lines.aggregate([
  {
    $match: {
      order_date: {
        $gte: "2020-01-10 00:00:00",
        $lte: "2020-01-15 23:59:59"
      }
    }
  },
  {
    $group: {
      _id: null,
      total_orders: { $sum: 1 }
    }
  },
  {
    $project: {
      _id: 0,
      total_orders: 1
    }
  }
])
